import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scannerObj = new Scanner(System.in);
        
        double semiPerimeter, side1, side2, side3, finalArea, calculationBuffer;
        
        System.out.println("Input the first side length: ");
        side1 = scannerObj.nextDouble();
        
        System.out.println("Input the second side length: ");
        side2 = scannerObj.nextDouble();
        
        System.out.println("Input the third side length: ");
        side3 = scannerObj.nextDouble();
        
        semiPerimeter = (side1 + side2 + side3) / 2;
        calculationBuffer = semiPerimeter * (semiPerimeter - side1) * (semiPerimeter - side2) * (semiPerimeter - side3);
        finalArea = Math.sqrt(calculationBuffer);
        
        System.out.println("Calculated Triangle Area: " + finalArea);
    }
}
